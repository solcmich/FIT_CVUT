def plot_pca(dataset):
    """
    Plots observed metrics for PCA on examind components count.
    """
    p = dataset[['preprocessor', 'f1', 'precision', 'recall']].groupby('preprocessor').mean()
    labels =  ['None', 'PCA(10)', 'PCA(100)', 'PCA(20)', 'PCA(40)','PCA(5)', 'PCA(60)', 'PCA(80)']
    x = np.arange(len(labels))
    width = 0.20
    fig, ax = plt.subplots()
    rects1 = ax.bar(x - width/2, p['f1'].values, width, label='f1')
    rects2 = ax.bar(x + width/2, p['precision'].values, width, label='precision')
    rects3 = ax.bar(x + 1.5*width, p['recall'].values, width, label='recall')
    ax.set_ylabel('Scores')
    ax.set_title('Scores for normalization and PCA')
    ax.set_xticks(x)
    ax.set_xticklabels(labels)
    ax.legend()
    x1,x2,y1,y2 = plt.axis()  
    plt.axis((x1,x2,0,1))
    fig.tight_layout()
    plt.show()
